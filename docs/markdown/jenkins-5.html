Estructura de un Pipeline declarativo

    agent any

}

Variables Globales

    VARIABLE = "mi variable"
pipeline{
    agent any
    stages{
        stage("Stage 1"){
            steps{
                echo "${VARIABLE}"
            }
        }
        stage("Stage 2"){
            steps{
                VARIABLE = "mi nueva variable"
                echo "${VARIABLE}"
            }
        }
    }
}

Variables Locales

Bloque script 
    pipeline{
    agent any
    stages{
        stage("Stage 1"){
            steps{
                script {
                    def VARIABLE = "mi variable"
                    echo "${VARIABLE}"
                }
            }
        }
        stage("Stage 2"){
            steps{
                echo "${VARIABLE}"
            }
        }
    }
}

Variables de Entorno

    pipeline{
    agent any
    environment{
        VARIABLE_ENTORNO = "mi variable entorno"
    }
    stages{
        stage("Stage 1"){
            steps{
                echo "${env.VARIABLE_ENTORNO}"
            }
        }
    }
}

Etapas Post ejecución

Pipelines Post 
    pipeline{
    agent any
    stages{
        stage("Stage 1"){
            steps{
                echo "Stage"
            }
        }
    }
    post { 
        always { 
            echo 'always'
        }
        failure{
            echo 'failure'
        }
    }
}
